# Drives
df -h == list drives in Gb

# Transmission Service 
sudo service transmission-daemon [start, stop, restart] == start stop or restart transmission

# Transmission CLI
transmission-remote --auth 'user:pass' -l == auth into transmission
transmission-remote -a "link" == add torrent
transmission-remote -l == download status
transmission-remote -t "1,2" -r == remove torrents with ids 1 and 2

# uwsigi
sudo systemctl status uwsgi.service = check server status
sudo systemctl stop/start/restart uwsgi.service
